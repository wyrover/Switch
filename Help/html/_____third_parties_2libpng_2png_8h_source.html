316 /*
317  * Y2K compliance in libpng:
318  * =========================
319  *
320  * March 29, 2012
321  *
322  * Since the PNG Development group is an ad-hoc body, we can't make
323  * an official declaration.
324  *
325  * This is your unofficial assurance that libpng from version 0.71 and
326  * upward through 1.5.10 are Y2K compliant. It is my belief that
327  * earlier versions were also Y2K compliant.
328  *
329  * Libpng only has two year fields. One is a 2-byte unsigned integer
330  * that will hold years up to 65535. The other holds the date in text
331  * format, and will hold years up to 9999.
332  *
333  * The integer is
334  * "png_uint_16 year" in png_time_struct.
335  *
336  * The string is
337  * "png_char time_buffer" in png_struct
338  *
339  * There are seven time-related functions:
340  * png.c: png_convert_to_rfc_1123() in png.c
341  * (formerly png_convert_to_rfc_1152() in error)
342  * png_convert_from_struct_tm() in pngwrite.c, called in pngwrite.c
343  * png_convert_from_time_t() in pngwrite.c
344  * png_get_tIME() in pngget.c
345  * png_handle_tIME() in pngrutil.c, called in pngread.c
346  * png_set_tIME() in pngset.c
347  * png_write_tIME() in pngwutil.c, called in pngwrite.c
348  *
349  * All handle dates properly in a Y2K environment. The
350  * png_convert_from_time_t() function calls gmtime() to convert from system
351  * clock time, which returns (year - 1900), which we properly convert to
352  * the full 4-digit year. There is a possibility that applications using
353  * libpng are not passing 4-digit years into the png_convert_to_rfc_1123()
354  * function, or that they are incorrectly passing only a 2-digit year
355  * instead of "year - 1900" into the png_convert_from_struct_tm() function,
356  * but this is not under our control. The libpng documentation has always
357  * stated that it works with 4-digit years, and the APIs have been
358  * documented as such.
359  *
360  * The tIME chunk itself is also Y2K compliant. It uses a 2-byte unsigned
361  * integer to hold the year, and can hold years as large as 65535.
362  *
363  * zlib, upon which libpng depends, is also Y2K compliant. It contains
364  * no date-related code.
365  *
366  * Glenn Randers-Pehrson
367  * libpng maintainer
368  * PNG Development Group
369  */

1169 #ifdef PNG_READ_ALPHA_MODE_SUPPORTED
1170 /* How the alpha channel is interpreted - this affects how the color channels of
1171  * a PNG file are returned when an alpha channel, or tRNS chunk in a palette
1172  * file, is present.
1173  *
1174  * This has no effect on the way pixels are written into a PNG output
1175  * datastream. The color samples in a PNG datastream are never premultiplied
1176  * with the alpha samples.
1177  *
1178  * The default is to return data according to the PNG specification: the alpha
1179  * channel is a linear measure of the contribution of the pixel to the
1180  * corresponding composited pixel. The gamma encoded color channels must be
1181  * scaled according to the contribution and to do this it is necessary to undo
1182  * the encoding, scale the color values, perform the composition and reencode
1183  * the values. This is the 'PNG' mode.
1184  *
1185  * The alternative is to 'associate' the alpha with the color information by
1186  * storing color channel values that have been scaled by the alpha. The
1187  * advantage is that the color channels can be resampled (the image can be
1188  * scaled) in this form. The disadvantage is that normal practice is to store
1189  * linear, not (gamma) encoded, values and this requires 16-bit channels for
1190  * still images rather than the 8-bit channels that are just about sufficient if
1191  * gamma encoding is used. In addition all non-transparent pixel values,
1192  * including completely opaque ones, must be gamma encoded to produce the final
1193  * image. This is the 'STANDARD', 'ASSOCIATED' or 'PREMULTIPLIED' mode (the
1194  * latter being the two common names for associated alpha color channels.)
1195  *
1196  * Since it is not necessary to perform arithmetic on opaque color values so
1197  * long as they are not to be resampled and are in the final color space it is
1198  * possible to optimize the handling of alpha by storing the opaque pixels in
1199  * the PNG format (adjusted for the output color space) while storing partially
1200  * opaque pixels in the standard, linear, format. The accuracy required for
1201  * standard alpha composition is relatively low, because the pixels are
1202  * isolated, therefore typically the accuracy loss in storing 8-bit linear
1203  * values is acceptable. (This is not true if the alpha channel is used to
1204  * simulate transparency over large areas - use 16 bits or the PNG mode in
1205  * this case!) This is the 'OPTIMIZED' mode. For this mode a pixel is
1206  * treated as opaque only if the alpha value is equal to the maximum value.
1207  *
1208  * The final choice is to gamma encode the alpha channel as well. This is
1209  * broken because, in practice, no implementation that uses this choice
1210  * correctly undoes the encoding before handling alpha composition. Use this
1211  * choice only if other serious errors in the software or hardware you use
1212  * mandate it; the typical serious error is for dark halos to appear around
1213  * opaque areas of the composited PNG image because of arithmetic overflow.
1214  *
1215  * The API function png_set_alpha_mode specifies which of these choices to use
1216  * with an enumerated 'mode' value and the gamma of the required output:
1217  */
1218 #define PNG_ALPHA_PNG 0 /* according to the PNG standard */
1219 #define PNG_ALPHA_STANDARD 1 /* according to Porter/Duff */
1220 #define PNG_ALPHA_ASSOCIATED 1 /* as above; this is the normal practice */
1221 #define PNG_ALPHA_PREMULTIPLIED 1 /* as above */
1222 #define PNG_ALPHA_OPTIMIZED 2 /* 'PNG' for opaque pixels, else 'STANDARD' */
1223 #define PNG_ALPHA_BROKEN 3 /* the alpha channel is gamma encoded */
1224 
1225 PNG_FP_EXPORT(227, void, png_set_alpha_mode, (png_structp png_ptr, int mode,
1226  double output_gamma));
1227 PNG_FIXED_EXPORT(228, void, png_set_alpha_mode_fixed, (png_structp png_ptr,
1228  int mode, png_fixed_point output_gamma));
1229 #endif

1231 #if defined(PNG_READ_GAMMA_SUPPORTED) || defined(PNG_READ_ALPHA_MODE_SUPPORTED)
1232 /* The output_gamma value is a screen gamma in libpng terminology: it expresses
1233  * how to decode the output values, not how they are encoded. The values used
1234  * correspond to the normal numbers used to describe the overall gamma of a
1235  * computer display system; for example 2.2 for an sRGB conformant system. The
1236  * values are scaled by 100000 in the _fixed version of the API (so 220000 for
1237  * sRGB.)
1238  *
1239  * The inverse of the value is always used to provide a default for the PNG file
1240  * encoding if it has no gAMA chunk and if png_set_gamma() has not been called
1241  * to override the PNG gamma information.
1242  *
1243  * When the ALPHA_OPTIMIZED mode is selected the output gamma is used to encode
1244  * opaque pixels however pixels with lower alpha values are not encoded,
1245  * regardless of the output gamma setting.
1246  *
1247  * When the standard Porter Duff handling is requested with mode 1 the output
1248  * encoding is set to be linear and the output_gamma value is only relevant
1249  * as a default for input data that has no gamma information. The linear output
1250  * encoding will be overridden if png_set_gamma() is called - the results may be
1251  * highly unexpected!
1252  *
1253  * The following numbers are derived from the sRGB standard and the research
1254  * behind it. sRGB is defined to be approximated by a PNG gAMA chunk value of
1255  * 0.45455 (1/2.2) for PNG. The value implicitly includes any viewing
1256  * correction required to take account of any differences in the color
1257  * environment of the original scene and the intended display environment; the
1258  * value expresses how to *decode* the image for display, not how the original
1259  * data was *encoded*.
1260  *
1261  * sRGB provides a peg for the PNG standard by defining a viewing environment.
1262  * sRGB itself, and earlier TV standards, actually use a more complex transform
1263  * (a linear portion then a gamma 2.4 power law) than PNG can express. (PNG is
1264  * limited to simple power laws.) By saying that an image for direct display on
1265  * an sRGB conformant system should be stored with a gAMA chunk value of 45455
1266  * (11.3.3.2 and 11.3.3.5 of the ISO PNG specification) the PNG specification
1267  * makes it possible to derive values for other display systems and
1268  * environments.
1269  *
1270  * The Mac value is deduced from the sRGB based on an assumption that the actual
1271  * extra viewing correction used in early Mac display systems was implemented as
1272  * a power 1.45 lookup table.
1273  *
1274  * Any system where a programmable lookup table is used or where the behavior of
1275  * the final display device characteristics can be changed requires system
1276  * specific code to obtain the current characteristic. However this can be
1277  * difficult and most PNG gamma correction only requires an approximate value.
1278  *
1279  * By default, if png_set_alpha_mode() is not called, libpng assumes that all
1280  * values are unencoded, linear, values and that the output device also has a
1281  * linear characteristic. This is only very rarely correct - it is invariably
1282  * better to call png_set_alpha_mode() with PNG_DEFAULT_sRGB than rely on the
1283  * default if you don't know what the right answer is!
1284  *
1285  * The special value PNG_GAMMA_MAC_18 indicates an older Mac system (pre Mac OS
1286  * 10.6) which used a correction table to implement a somewhat lower gamma on an
1287  * otherwise sRGB system.
1288  *
1289  * Both these values are reserved (not simple gamma values) in order to allow
1290  * more precise correction internally in the future.
1291  *
1292  * NOTE: the following values can be passed to either the fixed or floating
1293  * point APIs, but the floating point API will also accept floating point
1294  * values.
1295  */
1296 #define PNG_DEFAULT_sRGB -1 /* sRGB gamma and color space */
1297 #define PNG_GAMMA_MAC_18 -2 /* Old Mac '1.8' gamma and color space */
1298 #define PNG_GAMMA_sRGB 220000 /* Television standards--matches sRGB gamma */
1299 #define PNG_GAMMA_LINEAR PNG_FP_1 /* Linear */
1300 #endif

1302 /* The following are examples of calls to png_set_alpha_mode to achieve the
1303  * required overall gamma correction and, where necessary, alpha
1304  * premultiplication.
1305  *
1306  * png_set_alpha_mode(pp, PNG_ALPHA_PNG, PNG_DEFAULT_sRGB);
1307  * This is the default libpng handling of the alpha channel - it is not
1308  * pre-multiplied into the color components. In addition the call states
1309  * that the output is for a sRGB system and causes all PNG files without gAMA
1310  * chunks to be assumed to be encoded using sRGB.
1311  *
1312  * png_set_alpha_mode(pp, PNG_ALPHA_PNG, PNG_GAMMA_MAC);
1313  * In this case the output is assumed to be something like an sRGB conformant
1314  * display preceeded by a power-law lookup table of power 1.45. This is how
1315  * early Mac systems behaved.
1316  *
1317  * png_set_alpha_mode(pp, PNG_ALPHA_STANDARD, PNG_GAMMA_LINEAR);
1318  * This is the classic Jim Blinn approach and will work in academic
1319  * environments where everything is done by the book. It has the shortcoming
1320  * of assuming that input PNG data with no gamma information is linear - this
1321  * is unlikely to be correct unless the PNG files where generated locally.
1322  * Most of the time the output precision will be so low as to show
1323  * significant banding in dark areas of the image.
1324  *
1325  * png_set_expand_16(pp);
1326  * png_set_alpha_mode(pp, PNG_ALPHA_STANDARD, PNG_DEFAULT_sRGB);
1327  * This is a somewhat more realistic Jim Blinn inspired approach. PNG files
1328  * are assumed to have the sRGB encoding if not marked with a gamma value and
1329  * the output is always 16 bits per component. This permits accurate scaling
1330  * and processing of the data. If you know that your input PNG files were
1331  * generated locally you might need to replace PNG_DEFAULT_sRGB with the
1332  * correct value for your system.
1333  *
1334  * png_set_alpha_mode(pp, PNG_ALPHA_OPTIMIZED, PNG_DEFAULT_sRGB);
1335  * If you just need to composite the PNG image onto an existing background
1336  * and if you control the code that does this you can use the optimization
1337  * setting. In this case you just copy completely opaque pixels to the
1338  * output. For pixels that are not completely transparent (you just skip
1339  * those) you do the composition math using png_composite or png_composite_16
1340  * below then encode the resultant 8-bit or 16-bit values to match the output
1341  * encoding.
1342  *
1343  * Other cases
1344  * If neither the PNG nor the standard linear encoding work for you because
1345  * of the software or hardware you use then you have a big problem. The PNG
1346  * case will probably result in halos around the image. The linear encoding
1347  * will probably result in a washed out, too bright, image (it's actually too
1348  * contrasty.) Try the ALPHA_OPTIMIZED mode above - this will probably
1349  * substantially reduce the halos. Alternatively try:
1350  *
1351  * png_set_alpha_mode(pp, PNG_ALPHA_BROKEN, PNG_DEFAULT_sRGB);
1352  * This option will also reduce the halos, but there will be slight dark
1353  * halos round the opaque parts of the image where the background is light.
1354  * In the OPTIMIZED mode the halos will be light halos where the background
1355  * is dark. Take your pick - the halos are unavoidable unless you can get
1356  * your hardware/software fixed! (The OPTIMIZED approach is slightly
1357  * faster.)
1358  *
1359  * When the default gamma of PNG files doesn't match the output gamma.
1360  * If you have PNG files with no gamma information png_set_alpha_mode allows
1361  * you to provide a default gamma, but it also sets the ouput gamma to the
1362  * matching value. If you know your PNG files have a gamma that doesn't
1363  * match the output you can take advantage of the fact that
1364  * png_set_alpha_mode always sets the output gamma but only sets the PNG
1365  * default if it is not already set:
1366  *
1367  * png_set_alpha_mode(pp, PNG_ALPHA_PNG, PNG_DEFAULT_sRGB);
1368  * png_set_alpha_mode(pp, PNG_ALPHA_PNG, PNG_GAMMA_MAC);
1369  * The first call sets both the default and the output gamma values, the
1370  * second call overrides the output gamma without changing the default. This
1371  * is easier than achieving the same effect with png_set_gamma. You must use
1372  * PNG_ALPHA_PNG for the first call - internal checking in png_set_alpha will
1373  * fire if more than one call to png_set_alpha_mode and png_set_background is
1374  * made in the same read operation, however multiple calls with PNG_ALPHA_PNG
1375  * are ignored.
1376  */
